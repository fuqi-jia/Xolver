## Phase 2: RollbackUnionFind

**New files:** `src/theory/euf/RollbackUnionFind.h`, `.cpp`

```cpp
class RollbackUnionFind {
public:
    EufClassId addNode();
    EufClassId find(EufClassId x) const;      // NO path compression
    bool same(EufClassId a, EufClassId b) const;
    bool unite(EufClassId a, EufClassId b);   // union-by-size; returns true iff merged
    size_t snapshot() const;                  // trail_.size()
    void rollback(size_t snap);               // undo merges back to snap
    uint32_t classSize(EufClassId root) const;

private:
    std::vector<EufClassId> parent_;
    std::vector<uint32_t> size_;
    struct Change {
        EufClassId childRoot;
        EufClassId parentRoot;
        uint32_t oldParentSize;
        bool merged;
    };
    std::vector<Change> trail_;
};
```

**Key rules:**
- `find()` walks parent links iteratively (read-only).
- `unite(a,b)`: if `ra == rb`, push `{ra, rb, 0, false}` to trail and return false.
- If merge happens: attach smaller root to larger root; push `{rb, ra, oldRaSize, true}`.
- `rollback()`: pop trail entries; if `merged`, restore `parent_[childRoot] = childRoot` and `size_[parentRoot] = oldParentSize`.

**Unit test:** Rollback to various snapshots; verify `same()` returns correct results after rollback.

### 3.4 Merge algorithm (processes to FIXPOINT)

```cpp
MergeStatus IncrementalEGraph::merge(EufTermId a, EufTermId b, const MergeReason& reason) {
    ensureFreshSignatureTable();
    mergeQueue_.push_back({a, b, reason});
    return processToFixpoint();

```

`processToFixpoint()`:
```cpp
MergeStatus processToFixpoint() {
    while (!mergeQueue_.empty()) {
        auto m = mergeQueue_.front();
        mergeQueue_.pop_front();

        if (same(m.a, m.b)) continue;

        EufClassId ra = rep(m.a);
        EufClassId rb = rep(m.b);

        // sort check
        if (tm_.node(m.a).sort != tm_.node(m.b).sort &&
            tm_.node(m.a).sort != NullSort && tm_.node(m.b).sort != NullSort) {
            return MergeStatus::SortMismatch;
        }

        // collect affected parents BEFORE unite
        std::vector<EufTermId> affected = collectParents(ra);
        auto affectedB = collectParents(rb);
        affected.insert(affected.end(), affectedB.begin(), affectedB.end());

        // record merge BEFORE unite (explanation needs pre-merge roots)
        MergeRecord rec;
        rec.id = static_cast<MergeId>(mergeRecords_.size());
        rec.lhs = m.a; rec.rhs = m.b;
        rec.lhsRootBefore = ra; rec.rhsRootBefore = rb;
        rec.reason = m.reason;
        mergeRecords_.push_back(rec);

        // unite
        bool didMerge = uf_.unite(ra, rb);
        if (!didMerge) continue; // should not happen due to same() check above

        // merge members: attach smaller class members to larger class members
        // NOTE: use uf_.classSize to pick winner, or simply use ra as winner if ra < rb
        // Simpler: always attach rb members to ra (ra is the winner in our uf)
        memberTrail_.push_back({ra, rb, members_[ra].size()});
        members_[ra].insert(members_[ra].end(), members_[rb].begin(), members_[rb].end());
        // DO NOT clear members_[rb]; leave it stale but only root's members are read

        // enqueue affected parents for congruence check
        for (EufTermId p : affected) {
            congruenceQueue_.push_back(p);
        }

        // check true/false conflict immediately
        if (trueTerm_ != NullEufTerm && falseTerm_ != NullEufTerm && same(trueTerm_, falseTerm_)) {
            return MergeStatus::TrueFalseConflict;
        }

        // process congruence queue
        auto cstatus = processCongruenceQueue();
        if (cstatus != MergeStatus::Ok) return cstatus;
    }
    return MergeStatus::Ok;
}
```

### 3.5 Congruence queue

```cpp
MergeStatus processCongruenceQueue() {
    while (!congruenceQueue_.empty()) {
        EufTermId app = congruenceQueue_.front();
        congruenceQueue_.pop_front();

        if (tm_.node(app).args.empty()) continue; // not an application

        AppSignature sig = computeSignature(app);
        auto it = sigTable_.find(sig);

        if (it == sigTable_.end()) {
            sigTable_.emplace(sig, app);
            continue;
        }

        EufTermId existing = it->second;

        // stale check: recompute signature of stored term
        if (computeSignature(existing) != sig) {
            sigTable_[sig] = app; // overwrite stale
            continue;
        }

        if (!same(existing, app)) {
            // build congruence reason with argPairs
            MergeReason cr;
            cr.kind = MergeReasonKind::Congruence;
            cr.app1 = existing; cr.app2 = app;
            const auto& n1 = tm_.node(existing);
            const auto& n2 = tm_.node(app);
            for (size_t i = 0; i < n1.args.size(); ++i) {
                cr.argPairs.push_back({n1.args[i], n2.args[i]});
            }
            mergeQueue_.push_back({existing, app, cr});

            // recurse into merge processing (but through loop, not C recursion)
            auto ms = processToFixpoint();
            if (ms != MergeStatus::Ok) return ms;
            return MergeStatus::Ok; // queue consumed by recursive call
        }
    }
    return MergeStatus::Ok;
}
```

### 3.6 Signature table lazy rebuild

```cpp
void ensureFreshSignatureTable() {
    if (!sigTableDirty_) return;
    sigTable_.clear();
    for (EufTermId t = 0; t < tm_.termCount(); ++t) {
        if (tm_.node(t).args.empty()) continue;
        AppSignature sig = computeSignature(t);
        auto it = sigTable_.find(sig);
        if (it == sigTable_.end()) {
            sigTable_.emplace(sig, t);
        } else if (!same(it->second, t)) {
            // should not happen after clean rebuild, but enqueue to be safe
            mergeQueue_.push_back({it->second, t, MergeReason::congruence(it->second, t, sig)});
        }
    }
    sigTableDirty_ = false;
}
```

**When to call:** At the start of `merge()` and any public method that reads `sigTable_`.

### 3.7 Rollback

```cpp
void IncrementalEGraph::rollback(EGraphSnapshot snap) {
    // 1. rollback union-find
    uf_.rollback(snap.ufTrailSize);

    // 2. rollback mergeRecords
    mergeRecords_.resize(snap.mergeRecordSize);

    // 3. rollback members
    while (memberTrail_.size() > snap.memberTrailSize) {
        auto ch = memberTrail_.back();
        memberTrail_.pop_back();
        members_[ch.srcRoot].assign(
            members_[ch.dstRoot].begin() + ch.oldDstSize,
            members_[ch.dstRoot].end());
        members_[ch.dstRoot].resize(ch.oldDstSize);
    }

    // 4. rollback queues
    mergeQueue_.resize(snap.mergeQueueSize);
    congruenceQueue_.resize(snap.congruenceQueueSize);

    // 5. mark sigTable dirty
    sigTableDirty_ = true;
}
```

**Note:** `members_[srcRoot]` is NOT cleared during normal operation; it becomes stale but is only accessed via the current root. On rollback, we restore it.

### 3.8 Explanation

```cpp
ExplainResult IncrementalEGraph::explainEquality(EufTermId a, EufTermId b) {
    if (a == b) return {true, {}};
    if (!same(a, b)) return {false, {}};

    // Build temporary adjacency from active mergeRecords
    std::unordered_map<EufTermId, std::vector<std::pair<EufTermId, MergeId>>> adj;
    for (const auto& rec : mergeRecords_) {
        adj[rec.lhs].push_back({rec.rhs, rec.id});
        adj[rec.rhs].push_back({rec.lhs, rec.id});
    }

    // BFS
    std::unordered_map<EufTermId, EufTermId> parentTerm;
    std::unordered_map<EufTermId, MergeId> parentMerge;
    std::queue<EufTermId> q;
    std::unordered_set<EufTermId> visited;

    q.push(a); visited.insert(a); parentTerm[a] = a;

    while (!q.empty()) {
        EufTermId cur = q.front(); q.pop();
        auto it = adj.find(cur);
        if (it == adj.end()) continue;
        for (auto [next, mid] : it->second) {
            if (visited.count(next)) continue;
            visited.insert(next);
            parentTerm[next] = cur;
            parentMerge[next] = mid;
            if (next == b) {
                // reconstruct path
                std::vector<SatLit> reasons;
                std::unordered_set<uint64_t> activePairs;
                EufTermId node = b;
                while (node != a) {
                    MergeId mid = parentMerge[node];
                    auto er = explainMergeReason(mergeRecords_[mid].reason, activePairs);
                    if (!er.ok) return {false, {}};
                    appendUnique(reasons, er.reasons);
                    node = parentTerm[node];
                }
                return {true, std::move(reasons)};
            }
            q.push(next);
        }
    }
    return {false, {}}; // path not found — internal error
}
```

`explainMergeReason()` with cycle guard:
```cpp
ExplainResult explainMergeReason(const MergeReason& r,
                                 std::unordered_set<uint64_t>& activePairs) {
    if (r.kind == MergeReasonKind::AssertedEquality) {
        return {true, {r.assertedLit}};
    }
    std::vector<SatLit> out;
    for (auto [ai, bi] : r.argPairs) {
        if (ai == bi) continue;
        uint64_t pk = pairKey(ai, bi);
        if (!activePairs.insert(pk).second) return {false, {}}; // cycle
        auto sub = explainEquality(ai, bi, activePairs); // recursive with cycle guard
        activePairs.erase(pk);
        if (!sub.ok) return {false, {}};
        appendUnique(out, sub.reasons);
    }
    return {true, std::move(out)};
}
```

**Note:** `ExplainResult` already exists in `EufTypes.h`; reuse it.

### 4.4 pop()

```cpp
void EufSolver::pop(uint32_t n) {
    while (n-- && !scopeLimits_.empty()) {
        size_t limit = scopeLimits_.back();
        EGraphSnapshot snap = scopeSnapshots_.back();

        trail_.resize(limit);
        scopeLimits_.pop_back();
        scopeSnapshots_.pop_back();

        // rollback disequalities: remove all with level > current max level
        int maxLevel = trail_.empty() ? 0 : trail_.back().level;
        auto it = std::remove_if(disequalities_.begin(), disequalities_.end(),
            [maxLevel](const auto& d){ return d.level > maxLevel; });
        disequalities_.erase(it, disequalities_.end());

        // rollback egraph
        egraph_.rollback(snap);

        // clear pending conflict if it was generated at a level we're popping past
        if (pendingConflict_ && pendingConflictLevel_ > maxLevel) {
            pendingConflict_.reset();
            pendingConflictLevel_ = -1;
        }
    }
}
```

### 4.5 backtrackToLevel()

```cpp
void EufSolver::backtrackToLevel(int level) {
    // Find the snapshot at or before this level
    size_t bestSnapIdx = 0;
    for (size_t i = 0; i < scopeSnapshots_.size(); ++i) {
        if (static_cast<int>(scopeLimits_[i]) <= static_cast<int>(trail_.size())) {
            bestSnapIdx = i;
        }
    }

    // Remove trail entries above level
    auto it = std::remove_if(trail_.begin(), trail_.end(),
        [level](const auto& a){ return a.level > level; });
    trail_.erase(it, trail_.end());

    // Rollback egraph to the snapshot that corresponds to the scope before this level
    // Simpler: use the last scope snapshot whose limit is <= current trail size
    while (!scopeLimits_.empty() && scopeLimits_.back() > trail_.size()) {
        scopeLimits_.pop_back();
        scopeSnapshots_.pop_back();
    }

    if (!scopeSnapshots_.empty()) {
        egraph_.rollback(scopeSnapshots_.back());
    } else {
        egraph_.rollback(EGraphSnapshot{}); // reset to empty
    }

    // Remove disequalities above level
    auto dit = std::remove_if(disequalities_.begin(), disequalities_.end(),
        [level](const auto& d){ return d.level > level; });
    disequalities_.erase(dit, disequalities_.end());

    // Clear pending conflict (will be re-detected in check())
    pendingConflict_.reset();
    pendingConflictLevel_ = -1;
}
```

**Alternative simpler design:** `backtrackToLevel` is called by `CadicalTheoryPropagator` on notify_backtrack. In CDCL, backtrack goes to a specific decision level. The cleanest approach is to take a snapshot **at every decision level** (or at least maintain a mapping `level -> EGraphSnapshot`). For V1, we can snapshot at every `assertLit` that introduces a new level, or simply snapshot at `push()` boundaries and recompute.

**Recommended simplification for V1:**
- `push()` = `scopeLimits_.push_back(trail_.size()); scopeSnapshots_.push_back(egraph_.snapshot());`
- `pop(n)` = rollback to `scopeSnapshots_[scopeSnapshots_.size() - n]`.
- `backtrackToLevel(level)` = find the nearest scope limit ≤ current trail size after level filtering, and rollback to that snapshot. If no scope, rollback to empty.

This is slightly conservative (may rollback more than strictly necessary) but correct.

### 4.6 assertLit() — INCREMENTAL

```cpp
void EufSolver::assertLit(const TheoryAtomRecord& atom, bool value, int level, SatLit reason) {
    if (!std::holds_alternative<EufAtomPayload>(atom.payload)) return;

    const auto& payload = std::get<EufAtomPayload>(atom.payload);

    // Intern terms (monotonic)
    EufTermId lhs = termManager_.intern(payload.lhs, *coreIr_);
    EufTermId rhs = termManager_.intern(payload.rhs, *coreIr_);
    if (lhs == NullEufTerm || rhs == NullEufTerm) return;

    egraph_.ensureTerm(lhs);
    egraph_.ensureTerm(rhs);

    // Effective relation
    bool isEq = false;
    if (payload.rel == Relation::Eq) {
        isEq = value;
    } else if (payload.rel == Relation::Neq) {
        isEq = !value;
    } else {
        return; // unsupported
    }

    // Bool formula-position semantics:
    // Parser/Atomizer encodes Bool terms as (= term true)
    // If (= b true) is assigned false, then b = false
    if (payload.rel == Relation::Eq && payload.rhs == TrueSentinelExpr && !value) {
        EufTermId tFalse = termManager_.falseConstant();
        if (tFalse != NullEufTerm) {
            // Store the assignment, then merge b = false
            trail_.push_back({level, reason, atom, value});
            auto status = egraph_.merge(lhs, tFalse, MergeReason::asserted(reason));
            if (status == MergeStatus::TrueFalseConflict) {
                auto er = egraph_.explainEquality(lhs, tFalse);
                if (er.ok) {
                    for (auto& lit : er.reasons) lit = lit.negated();
                    pendingConflict_ = TheoryConflict{std::move(er.reasons)};
                    pendingConflictLevel_ = level;
                }
            }
            return;
        }
    }

    trail_.push_back({level, reason, atom, value});

    if (isEq) {
        auto status = egraph_.merge(lhs, rhs, MergeReason::asserted(reason));
        if (status == MergeStatus::TrueFalseConflict) {
            auto er = egraph_.explainEquality(egraph_.trueTerm(), egraph_.falseTerm());
            if (er.ok) {
                for (auto& lit : er.reasons) lit = lit.negated();
                pendingConflict_ = TheoryConflict{std::move(er.reasons)};
                pendingConflictLevel_ = level;
            }
        }
    } else {
        disequalities_.push_back({lhs, rhs, reason, level});
        if (egraph_.same(lhs, rhs)) {
            auto er = egraph_.explainEquality(lhs, rhs);
            if (er.ok) {
                for (auto& lit : er.reasons) lit = lit.negated();
                er.reasons.push_back(reason.negated());
                pendingConflict_ = TheoryConflict{std::move(er.reasons)};
                pendingConflictLevel_ = level;
            }
        }
    }
}
```

### 4.7 check()

```cpp
TheoryCheckResult EufSolver::check(TheoryLemmaDatabase& /*lemmaDb*/) {
    // 1. Pending conflict from incremental assertLit
    if (pendingConflict_) {
        TheoryConflict c = *pendingConflict_;
        pendingConflict_.reset();
        pendingConflictLevel_ = -1;
        return TheoryCheckResult::mkConflict(std::move(c));
    }

    // 2. Check true/false invariant (defense in depth)
    EufTermId tTrue = termManager_.trueConstant();
    EufTermId tFalse = termManager_.falseConstant();
    if (tTrue != NullEufTerm && tFalse != NullEufTerm && egraph_.same(tTrue, tFalse)) {
        auto er = egraph_.explainEquality(tTrue, tFalse);
        if (er.ok) {
            for (auto& lit : er.reasons) lit = lit.negated();
            return TheoryCheckResult::mkConflict(TheoryConflict{std::move(er.reasons)});
        }
        return TheoryCheckResult::unknown();
    }

    // 3. Check disequalities
    for (const auto& d : disequalities_) {
        if (egraph_.same(d.lhs, d.rhs)) {
            auto er = egraph_.explainEquality(d.lhs, d.rhs);
            if (er.ok) {
                for (auto& lit : er.reasons) lit = lit.negated();
                er.reasons.push_back(d.reason.negated());
                return TheoryCheckResult::mkConflict(TheoryConflict{std::move(er.reasons)});
            }
            return TheoryCheckResult::unknown();
        }
    }

    // 4. Propagation TODO (Revision C)
    // if (enablePropagation_) { ... }

    return TheoryCheckResult::consistent();
}
```

## Red Lines

1. **No path compression** in `RollbackUnionFind`.
2. **`merge()` processes to fixpoint** — returns only after mergeQueue + congruenceQueue empty.
3. **Stale signatures detected** — `findFresh()` recomputes `computeSignature(storedTerm)`.
4. **Signature table lazy rebuild on backtrack** — `sigTableDirty_` cleared by `ensureFreshSignatureTable()`.
5. **Members rollback** — `memberTrail_` with `MemberChange{dst, src, oldDstSize}`; rollback does `resize`.
6. **MergeRecords rollback** — `resize()` on snapshot.
7. **Queues cleared on rollback** — `mergeQueue_` and `congruenceQueue_` resized to snapshot sizes.
